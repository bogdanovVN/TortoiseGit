namespace Powers of two
{
    class Powers of two
    {
        static void Main(string[] args)
        {
            var n = int.Parse(Console.ReadLine());

            var number = 1;
            for (int exponent = 0; exponent <= n; exponent++)
            {
                Console.WriteLine(number);
                number *= 2;
            }
        }
    }
}
